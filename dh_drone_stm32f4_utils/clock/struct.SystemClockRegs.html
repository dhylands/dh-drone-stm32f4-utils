pub fn handler(reg: Regs, thr_init: ThrsInit) {
    let thr = thr::init(thr_init);
    let sysclk = WeActSystemClock::init();

    let sysclk_regs = SystemClockRegs::init(
        reg.flash_acr,
        reg.pwr_cr,
        reg.rcc_cr,
        reg.rcc_pllcfgr,
        reg.rcc_cfgr,
        reg.rcc_cir.into_copy(),
        reg.rcc_apb1enr,
        thr.rcc,
        &sysclk,
    );
    sysclk_regs.raise_system_frequency().root_wait();
